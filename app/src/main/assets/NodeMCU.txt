#include <ArduinoJson.h>
#include <ESP8266WiFi.h>
#include <WiFiClient.h>
#include <ESP8266WebServer.h>


const char* ssid = "Farmbot";         // Replace with your desired access point SSID
const char* password = "11223344"; // Replace with your desired access point password

ESP8266WebServer server(80);

String data;

void handleRoot() {
  DynamicJsonDocument doc(128);
  doc["temperature"] = data;

  String response;
  serializeJson(doc, response);

  server.send(200, "application/json", response);
}




void setup() {
  Serial.begin(115200);


  WiFi.softAP(ssid, password); // Set up NodeMCU as an access point

  IPAddress IP = WiFi.softAPIP();
  Serial.print("Access Point IP address: ");
  Serial.println(IP);

  server.on("/GetData", handleRoot);
  server.begin();
  Serial.println("Server started!");

}

void loop() {
  server.handleClient();
  if(Serial.available()>0){
    data = Serial.readStringUntil('\n');
    Serial.println(data);
  }

  delay (100);

}
